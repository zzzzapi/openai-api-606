Node.js调用OpenAI API示例代码


// OpenAI API调用示例 - GPT-4模型
const axios = require('axios');

const API_KEY = 'your_api_key_here'; // 替换为您的API密钥
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions'; // OpenAI API中转地址

async function callOpenAI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'gpt-4', // 使用GPT-4模型
        messages: [
          { role: 'system', content: '你是一个专业的编程助手' },
          { role: 'user', content: '请用JavaScript写一个简单的Todo应用' }
        ],
        temperature: 0.7,
        max_tokens: 800
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    console.log(response.data.choices[0].message.content);
  } catch (error) {
    console.error('调用OpenAI API出错:', error.response?.data || error.message);
  }
}

callOpenAI();
